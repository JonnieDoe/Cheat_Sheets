$ git checkout source_branch <paths>...

    $ git branch
    * master
    test_integration

    $ git checkout test_integration app/models/avatar.rb db/migrate/20090223104419_create_avatars.rb

    $ git status
    # On branch master
    # Changes to be committed:

    #
    #	new file:   app/models/avatar.rb

    #
    
    $ git commit -m "'Merge' avatar code from 'test_integration' branch"
